export const title = 'rotate'
export const description = 'Utilities for rotating elements.'

## rotate

Utilities for rotating elements.

| Utility                | Styles                                                                      |
| ---------------------- | --------------------------------------------------------------------------- |
| `rotate.none`          | `rotate: none;`                                                             |
| `rotate[*<value>*]`    | `rotate: *<number>*deg;`                                                    |
| `-rotate[*<value>*]`   | `rotate: calc(*<number>*deg * -1);`                                         |
| `rotate.x[*<value>*]`  | `transform: rotateX(*<number>*deg) var(--tw-rotate-y);`                     |
| `-rotate.x[*<value>*]` | `transform: rotateX(-*<number>*deg) var(--tw-rotate-y);`                    |
| `rotate.y[*<value>*]`  | `transform: var(--tw-rotate-x) rotateY(*<number>*deg);`                     |
| `-rotate.y[*<value>*]` | `transform: var(--tw-rotate-x) rotateY(-*<number>*deg);`                    |
| `rotate.z[*<value>*]`  | `transform: var(--tw-rotate-x) var(--tw-rotate-y) rotateZ(*<number>*deg);`  |
| `-rotate.z[*<value>*]` | `transform: var(--tw-rotate-x) var(--tw-rotate-y) rotateZ(-*<number>*deg);` |
