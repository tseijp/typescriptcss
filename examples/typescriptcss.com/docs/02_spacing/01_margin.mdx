export const title = 'margin'
export const description = "Utilities for controlling an element's margin."

## margin

| Utility          | Styles                                                    |
| ---------------- | --------------------------------------------------------- |
| `m[*<value>*]`   | `margin: calc(var(--spacing) * *<number>*);`              |
| `margin.auto`    | `margin: auto;`                                           |
| `margin['1px']`  | `margin: 1px;`                                            |
| `margin['-1px']` | `margin: -1px;`                                           |
| `mx[*<value>*]`  | `margin-inline: calc(var(--spacing) * *<number>*);`       |
| `mx.auto`        | `margin-inline: auto;`                                    |
| `mx['1px']`      | `margin-inline: 1px;`                                     |
| `mx['-1px']`     | `margin-inline: -1px;`                                    |
| `my[*<value>*]`  | `margin-block: calc(var(--spacing) * *<number>*);`        |
| `my.auto`        | `margin-block: auto;`                                     |
| `my['1px']`      | `margin-block: 1px;`                                      |
| `my['-1px']`     | `margin-block: -1px;`                                     |
| `ms[*<value>*]`  | `margin-inline-start: calc(var(--spacing) * *<number>*);` |
| `ms.auto`        | `margin-inline-start: auto;`                              |
| `ms['1px']`      | `margin-inline-start: 1px;`                               |
| `ms['-1px']`     | `margin-inline-start: -1px;`                              |
| `me[*<value>*]`  | `margin-inline-end: calc(var(--spacing) * *<number>*);`   |
| `me.auto`        | `margin-inline-end: auto;`                                |
| `me['1px']`      | `margin-inline-end: 1px;`                                 |
| `me['-1px']`     | `margin-inline-end: -1px;`                                |
| `mbs[*<value>*]` | `margin-block-start: calc(var(--spacing) * *<number>*);`  |
| `mt.auto`        | `margin-block-start: auto;`                               |
| `mt['1px']`      | `margin-block-start: 1px;`                                |
| `mt['-1px']`     | `margin-block-start: -1px;`                               |
| `mbe[*<value>*]` | `margin-block-end: calc(var(--spacing) * *<number>*);`    |
| `mb.auto`        | `margin-block-end: auto;`                                 |
| `mb['1px']`      | `margin-block-end: 1px;`                                  |
| `mb['-1px']`     | `margin-block-end: -1px;`                                 |
| `mt[*<value>*]`  | `margin-top: calc(var(--spacing) * *<number>*);`          |
| `mr[*<value>*]`  | `margin-right: calc(var(--spacing) * *<number>*);`        |
| `mr.auto`        | `margin-right: auto;`                                     |
| `mr['1px']`      | `margin-right: 1px;`                                      |
| `mr['-1px']`     | `margin-right: -1px;`                                     |
| `mb[*<value>*]`  | `margin-bottom: calc(var(--spacing) * *<number>*);`       |
| `ml[*<value>*]`  | `margin-left: calc(var(--spacing) * *<number>*);`         |
| `ml.auto`        | `margin-left: auto;`                                      |
| `ml['1px']`      | `margin-left: 1px;`                                       |
| `ml['-1px']`     | `margin-left: -1px;`                                      |
