export const title = 'mask-image'
export const description = "Utilities for controlling an element's mask image."

## mask-image

Utilities for controlling an element's mask image.

| Utility                       | Styles                                                                                                                                                                                                                                                      |
| ----------------------------- | ----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| `mask[*<value>*]`             | `mask-image: *<value>*;`                                                                                                                                                                                                                                    |
| `mask.none`                   | `mask-image: none;`                                                                                                                                                                                                                                         |
| `mask.linear[*<value>*]`      | `mask-image: linear-gradient(*<number>*deg, black var(--tw-mask-linear-from)), transparent var(--tw-mask-linear-to));`                                                                                                                                      |
| `-mask.linear[*<value>*]`     | `mask-image: linear-gradient(calc(*<number>*deg * -1), black var(--tw-mask-linear-from)), transparent var(--tw-mask-linear-to));`                                                                                                                           |
| `mask.linear.from[*<value>*]` | `mask-image: linear-gradient(var(--tw-mask-linear-position), black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-linear-to));`                                                                                                               |
| `mask.linear.to[*<value>*]`   | `mask-image: linear-gradient(var(--tw-mask-linear-position), black var(--tw-mask-linear-from), transparent calc(var(--spacing) * *<number>*));`                                                                                                             |
| `mask.t.from[*<value>*]`      | `mask-image: linear-gradient(to top, black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-top-to));`                                                                                                                                          |
| `mask.t.to[*<value>*]`        | `mask-image: linear-gradient(to top, black var(--tw-mask-top-from), transparent calc(var(--spacing) * *<number>*));`                                                                                                                                        |
| `mask.r.from[*<value>*]`      | `mask-image: linear-gradient(to right, black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-right-to));`                                                                                                                                      |
| `mask.r.to[*<value>*]`        | `mask-image: linear-gradient(to right, black var(--tw-mask-right-from), transparent calc(var(--spacing) * *<number>*));`                                                                                                                                    |
| `mask.b.from[*<value>*]`      | `mask-image: linear-gradient(to bottom, black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-bottom-to));`                                                                                                                                    |
| `mask.b.to[*<value>*]`        | `mask-image: linear-gradient(to bottom, black var(--tw-mask-bottom-from), transparent calc(var(--spacing) * *<number>*));`                                                                                                                                  |
| `mask.l.from[*<value>*]`      | `mask-image: linear-gradient(to left, black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-left-to));`                                                                                                                                        |
| `mask.l.to[*<value>*]`        | `mask-image: linear-gradient(to left, black var(--tw-mask-left-from), transparent calc(var(--spacing) * *<number>*));`                                                                                                                                      |
| `mask.y.from[*<value>*]`      | `mask-image: linear-gradient(to top, black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-top-to)), linear-gradient(to bottom, black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-bottom-to));mask-composite: intersect;`     |
| `mask.y.to[*<value>*]`        | `mask-image: linear-gradient(to top, black var(--tw-mask-top-from), transparent calc(var(--spacing) * *<number>*)), linear-gradient(to bottom, black var(--tw-mask-bottom-from), transparent calc(var(--spacing) * *<number>*));mask-composite: intersect;` |
| `mask.x.from[*<value>*]`      | `mask-image: linear-gradient(to right, black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-right-to)), linear-gradient(to left, black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-left-to));mask-composite: intersect;`     |
| `mask.x.to[*<value>*]`        | `mask-image: linear-gradient(to right, black var(--tw-mask-right-from), transparent calc(var(--spacing) * *<number>*)), linear-gradient(to left, black var(--tw-mask-left-from), transparent calc(var(--spacing) * *<number>*));mask-composite: intersect;` |
| `mask.radial[*<value>*]`      | `mask-image: radial-gradient(*<value>*);`                                                                                                                                                                                                                   |
| `mask.radial.from[*<value>*]` | `mask-image: radial-gradient(var(--tw-mask-radial-shape) var(--tw-mask-radial-size) at var(--tw-mask-radial-position), black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-radial-to));`                                                     |
| `mask.radial.to[*<value>*]`   | `mask-image: radial-gradient(var(--tw-mask-radial-shape) var(--tw-mask-radial-size) at var(--tw-mask-radial-position), black var(--tw-mask-radial-from), transparent calc(var(--spacing) * *<number>*));`                                                   |
| `mask.conic[*<value>*]`       | `mask-image: conic-gradient(from *<number>*deg, black var(--tw-mask-conic-from), transparent var(--tw-mask-conic-to));`                                                                                                                                     |
| `-mask.conic[*<value>*]`      | `mask-image: conic-gradient(from calc(*<number>*deg * -1), black var(--tw-mask-conic-from), transparent var(--tw-mask-conic-to));`                                                                                                                          |
| `mask.conic.from[*<value>*]`  | `mask-image: conic-gradient(from var(--tw-mask-conic-position), black calc(var(--spacing) * *<number>*), transparent var(--tw-mask-conic-to));`                                                                                                             |
| `mask.conic.to[*<value>*]`    | `mask-image: conic-gradient(from var(--tw-mask-conic-position), black var(--tw-mask-conic-from), transparent calc(var(--spacing) * *<number>*));`                                                                                                           |
